1. Reverse of a Number


        // Write a program to print the reverse of a number
        // input: 218
        // output: 812

        // input:-210
        // output: -12
        function reverse(n){
            s = 0;
            while (n !== 0){
                r = n % 10;
                s = s * 10 + r;
                n = n / 10;
                n = Math.trunc(n);
            }
            return s;
        }
        //console.log(reverse(-210));

2. Min, Max, and Rest of the Numbers


        // Write a program to print min, max, and followed by rest of the numbers
        // input: 13675
        // output: 17365
        function mmr(s){
            min = Math.min(...String(s).split(""));
            max = Math.max(...String(s).split(""));
            rs = '' + min + max;
            for(var i of String(s)){
                if(!rs.includes(i)){
                    rs += i;
                }
            }
            return rs;
        }
        //console.log(mmr(12345));

3. Palindrome Check


        // Write a program to check whether a number is palindrome or not.
        // input: 121
        // output: 121 is palindrome

        // input: 467
        // output: 467 is not a palindrome
        function palindrome(n){
            s = 0;
            num = n;
            while (n != 0){
                r = n % 10;
                s = s * 10 + r;
                n = n / 10;
                n = Math.trunc(n);
            }
            return s == num ? "Palindrome" : "Not Palindrome";
        }
        //console.log(palindrome(128));

4. Unique Number Check


        // Write a program to print true if the number is unique else print false if it contains duplicates
        // input: 1211
        // output: False

        // input: 453
        // output: True
        function unique(s){
            n1 = new Set(String(s)).size;
            n2 = String(s).length;
            return n1 == n2 ? "unique" : "not unique";
        }
        //console.log(unique(1234));

5. Factorial Calculation


        // Write a program to print the below expression for the given number (Factorial)
        // input: 5
        // 5*4*3*2*1=120
        function factorial(n){
            r = 1;
            for (i = 1; i <= n; i++){
                r *= i;
            }
            return r;
        }
        console.log(factorial(5));
