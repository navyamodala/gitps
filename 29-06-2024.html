Pattern 1: Alphabetic Pattern

Pattern:

        
function alphabeticPattern1(rows) {
    var result = '';
    var currentCharCode = 97; // ASCII code for 'a'
    for (var i = 1; i <= rows; i++) {
        var line = '';
        for (var j = 1; j <= i; j++) {
            line += String.fromCharCode(currentCharCode) + ' ';
            currentCharCode++;
        }
        result += line.trim() + '\\n';
    }
    return result;
}

// Example usage:
var rows1 = prompt('Enter number of rows for Pattern 1:');
var pattern1 = alphabeticPattern1(parseInt(rows1));
console.log(pattern1);

Pattern 2: Mixed Case Alphabetic Pattern

Pattern:

        
function alphabeticPattern2(rows) {
    var result = '';
    var currentCharCode = 65; // ASCII code for 'A'
    var count = 0;
    for (var i = 1; i <= rows; i++) {
        var line = '';
        for (var j = 1; j <= i; j++) {
            line += String.fromCharCode(currentCharCode) + ' ';
            currentCharCode++;
        }
        result += line.trim() + '\\n';
        currentCharCode += count + 1;
        count++;
    }
    return result;
}

// Example usage:
var rows2 = prompt('Enter number of rows for Pattern 2:');
var pattern2 = alphabeticPattern2(parseInt(rows2));
console.log(pattern2);
