Time Tasks

Task 1: Perfect Times (Minutes = 00)

        
function findPerfectTimes(timearr) {
    for (var i of timearr) {
        for (var j in i) {
            var x = i[j].split(":");
            if (x[1] === "00") {
                console.log(i);
            }
        }
    }
}

// Example usage:
var timearr1 = [{"Time1": "10:34"}, {"Time2": "12:00"}, {"Time3": "02:30"}, {"Time4": "12:45"}];
findPerfectTimes(timearr1);

Task 2: Non-Perfect Times (Minutes !== 00)

        
function findNonPerfectTimes(timearr) {
    for (var i of timearr) {
        for (var j in i) {
            var x = i[j].split(":");
            if (x[1] !== "00") {
                console.log(i);
            }
        }
    }
}

// Example usage:
var timearr2 = [{"Time1": "10:34"}, {"Time2": "12:00"}, {"Time3": "02:30"}, {"Time4": "12:45"}];
findNonPerfectTimes(timearr2);

Task 3: Delayed Times (Minutes Calculation)

        
function calculateDelayedTimes(timearr) {
    for (var i of timearr) {
        for (var j in i) {
            var x = i[j].split(":");
            if (x[1] !== "00") {
                var delayMinutes = 60 - parseInt(x[1]);
                console.log(j + ": " + delayMinutes + " min");
            }
        }
    }
}

// Example usage:
var timearr3 = [{"Time1": "10:34"}, {"Time2": "12:00"}, {"Time3": "02:30"}, {"Time4": "12:45"}];
calculateDelayedTimes(timearr3);
