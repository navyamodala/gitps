        //1.how to print prime numbers
        function prime(n){          
            p=false
            for(var i=2;i<n;i++){
                if(n%i===0){
                    p=true;
                    break;
                }
            }
            if (!p){
                return "prime"
            }
            return "not a prime number"
        }

        function nearestprime(n){
            if(prime(n)){
                return n
            }
            l=[]
            x=2;
            while (x!=0){
                if(prime(x)){
                    l.push(x)
                    if(x>n){
                    break;
                }
                }
               
                x++
            }
            len=l.length
            return  n-l[len-2]<l[len-1]-n? l[len-2] :l[len-1]
        }
        
        function sumofprime(n){
            sum=0
            for (j  of String(n)){
                if(prime(+j)){
