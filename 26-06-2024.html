JavaScript Patterns

Pattern 1:

        
function pattern1(n) {
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var k = i - 1; k >= 1; k--) {
            s += " ";
        }
        for (var j = n - i + 1; j >= 1; j--) {
            s += "* ";
        }
        console.log(s);
    }
}
pattern1(5);

Pattern 2:

        
function pattern2(n) {
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var j = 1; j <= i; j++) {
            if (i % 2 == 1) {
                if (j % 2 == 1) {
                    s += '1 ';
                } else {
                    s += '0 ';
                }
            } else {
                if (j % 2 == 0) {
                    s += '1 ';
                } else {
                    s += '0 ';
                }
            }
        }
        console.log(s);
    }
}
pattern2(4);

Pattern 3:

        
function pattern3(n) {
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var j = 1; j <= n; j++) {
            if (i == 1 || i == n) {
                s += "* ";
            } else {
                j == 1 || j == n ? s += "* " : s += "  ";
            }
        }
        console.log(s);
    }
}
pattern3(5);

Pattern 4:

        
function pattern4(n) {
    var num = 1;
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var j = 1; j <= i; j++) {
            s += num + " ";
            num += 2;
        }
        console.log(s);
    }
}
pattern4(4);

function pattern5(n) {
    var num = 1;
    var ar = [];
    for (var i = 1; i <= n; i++) {
        var s = [];
        for (var j = 1; j <= n; j++) {
            if (j < i) {
                s.push(" ");
            } else {
                s.push(num);
                num += 2;
            }
        }
        ar.push(s);
    }
  
    for (var i = 0; i < n; i++) {
        var r = '';
        for (var j = 0; j < n; j++) {
            r += ar[j][i] + ' ';
        }
        console.log(r);
    }
}
pattern5(4);
