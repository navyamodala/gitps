JavaScript Functions Example

underscore Function:

//to remove duplicates and insert _ at the duplicate place
function underscore(arr){
    for(var i=0;i<arr.length;i++){
        if(arr[i] === arr[i+1]){
            arr.splice(i,1);
            i--;
        }
    }
    return arr;
}

checkorder Function:

//to check whether an array is in sorted order or not
//[2,4,5,6]--returns ascending order
//[7,5,4,3]--returns descending order
function checkorder(arr){
    var asc = true;
    var desc = true;
    for(var i=0;i<arr.length-1;i++){
        for(var j=i+1;j<arr.length;j++){
            if(asc !== false){
                if(arr[i] > arr[j]){
                    asc = false;
                }
            }
            if(desc !== false){
                if(arr[i] < arr[j]){
                    desc = false;
                }
            }
        }
    }
    return asc ? "array is in ascending order" : desc ? "descending order" : "not in order";
}

Example Usage:

// Example usage of checkorder function
console.log(checkorder([2, 4, 5, 6])); // Output: "array is in ascending order"
console.log(checkorder([7, 5, 4, 3])); // Output: "descending order"
