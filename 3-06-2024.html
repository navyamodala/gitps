Task 2: Finding Min and Max in Number Sequence


// Task 2: Finding the minimum and maximum numbers and identifying which is found first and second
// i/p: enter a number : 123456789
// o/p: min is found first and max is found second
function minmaxfound(n) {
    n = String(n).split("")
    min = Math.min(...n)
    max = Math.max(...n)
    if (n.indexOf(min) < n.indexOf(max)) {
        return "min is found first and max is second"
    }
    return "Max is found first and min is second"
}
