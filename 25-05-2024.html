        //1.write a program to print reverse of a number
        function reverse(n){
            s = 0;
            while(n > 0){
                r = n % 10;
                s = s * 10 + r;
                n = n / 10;
                n = Math.floor(n);
            }
            return s;
        }

        //2. write a program to check whether input 2 is available in input1
        // input1: 108
        // input2: 8
        // true
        // input1: 535
        // input2: 6
        // false
        function checker(n, v){
            for(i of String(n)){
                if(v == +i){
                    return true;
                }
            }
            return false;
        }

        //write a program to check whether a number is even or odd using switch statement
        function usingcase(n){
            switch(n % 2 == 0){
                case true:
                    return "even number";
                case false:
                    return "odd number";
            }
        }
