countAndSortCharacters Function:

        
function countAndSortCharacters(inputString) {
    // Convert inputString to lowercase to ignore case sensitivity
    inputString = inputString.toLowerCase();

    // Object to store character counts
    let charCount = {};

    // Count occurrences of each character
    for (let char of inputString) {
        if (charCount[char]) {
            charCount[char]++;
        } else {
            charCount[char] = 1;
        }
    }

    // Sort characters alphabetically
    let sortedChars = Object.keys(charCount).sort();

    // Generate output in the required format
    let output = '';
    for (let char of sortedChars) {
        output += `${char} - ${charCount[char]}\n`;
    }

    return output;
}

// Example usage:
let countResult = countAndSortCharacters("Good Bye");
console.log(countResult);  // This logs the output to console

countAndSortByFrequency Function:

        
function countAndSortByFrequency(inputString) {
    // Convert inputString to lowercase to ignore case sensitivity
    inputString = inputString.toLowerCase();

    // Object to store character counts
    let charCount = {};

    // Count occurrences of each character
    for (let char of inputString) {
        if (charCount[char]) {
            charCount[char]++;
        } else {
            charCount[char] = 1;
        }
    }

    // Sort characters by frequency (descending order)
    let sortedChars = Object.keys(charCount).sort((a, b) => charCount[b] - charCount[a]);

    // Generate output in the required format
    let output = '';
    for (let char of sortedChars) {
        output += `${char} - ${charCount[char]}\n`;
    }

    return output;
}

// Example usage:
let frequencyResult = countAndSortByFrequency("Good Bye");
console.log(frequencyResult);  // This logs the output to console

outputOrder Function:

        
function outputOrder(inputString, pattern) {
    let output = '';
    for (let char of pattern) {
        if (inputString.includes(char)) {
            output += char;
        }
    }
    return output;
}

// Example usage:
let orderedOutput = outputOrder("programming", "oai-iao");
console.log(orderedOutput);  // This logs the output to console
