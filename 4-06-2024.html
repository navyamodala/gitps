1. Highest Occurrence of Duplicate Number


        //1.write a program to print the highest occurrence of the duplicate number.
        var arr = [17, 77, 4, 17, 3, 3, 4, 4, 100, 4];
        var empty = [];
        var dupcount = 0;
        var dupval = 0;
        for (var i of arr) {
            if (!empty.includes(i)) {
                empty.push(i);
            }
        }

        for (var i of empty) {
            var c = 0;
            for (var j of arr) {
                if (i == j) {
                    c += 1;
                }
            }
            if (c > dupcount) {
                dupcount = c;
                dupval = i;
            }
        }

        console.log(dupval);

2. Second Highest Duplicate Number


        //2.write a program to print the second highest duplicate number.
        //o/p:4;
        var arr = [17, 77, 4, 17, 3, 3, 4, 4, 100, 4];
        var empty = [];
        var dfmaxc = 0;
        var dfmaxv = 0;
        var dsmaxc = 0;
        var dsmaxv = 0;
        for (var i of arr) {
            if (!empty.includes(i)) {
                empty.push(i);
            }
        }
        for (var i of empty) {
            var c = 0;
            for (var j of arr) {
                if (i == j) {
                    c += 1;
                }
            }
            if (c > 1 && c > dfmaxc) {
                dsmaxc = dfmaxc;
                dsmaxv = dfmaxv;
                dfmaxc = c;
                dfmaxv = i;
            } else if (c > 1 && c != dfmaxc && c > dsmaxc) {
                dsmaxc = c;
                dsmaxv = i;
            }
        }
        console.log(dsmaxv);
