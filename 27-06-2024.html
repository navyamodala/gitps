JavaScript Patterns

Pattern 1:

        
function pattern1(n) {
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var j = 1; j <= i; j++) {
            s += "* ";
        }
        console.log(s);
    }
}
pattern1(5);

Pattern 2:

        
function pattern2(n) {
    for (var i = n; i >= 1; i--) {
        var s = '';
        for (var j = 1; j <= i; j++) {
            s += "* ";
        }
        console.log(s);
    }
}
pattern2(5);

Pattern 3:

        
function pattern3(n) {
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var j = 1; j <= n - i; j++) {
            s += "  ";
        }
        for (var k = 1; k <= i; k++) {
            s += "* ";
        }
        console.log(s);
    }
}
pattern3(5);

Pattern 4:

        
function pattern4(n) {
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var j = 1; j <= n - i; j++) {
            s += "  ";

        for (var k = 1; k <= 2 * i - 1; k++) {

        console.log(s);
    }
}
pattern4(5);

Pattern 5:

        
function pattern5(n) {
    for (var i = 1; i <= n; i++) {
        var s = '';
        for (var j = 1; j <= n - i; j++) {
            s += "  ";
        }
        for (var k = 1; k <= i; k++) {
            s += "* ";
        }
        for (var m = i - 1; m >= 1; m--) {
            s += "* ";
        }
        console.log(s);
    }
}
pattern5(5);
