//1.to print even numbers up to the range
function even(n){
    s = '';
    for(var i = 1; i <= n; i++){
        if(i % 2 == 0){
            s += i + " ";
        }
    }
    return s;
}

//2.write a Program to print sum of odd numbers in the fibonacci series
function fibonacci(n){
    var a = 0;
    var b = 1;
    var s = 0;
    while(a < n){
        if(a % 2 == 1){
            s += a;
        }
        var c = a + b;
        a = b;
        b = c;
    }
    return s;
}
