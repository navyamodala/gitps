Number Pattern in JavaScript

Pattern:

        
function numberPattern(rows) {
    var num = 2;
    var result = '';
    for (var i = 1; i <= rows; i++) {
        var line = '';
        for (var j = 1; j <= i; j++) {
            line += num + ' ';
            num += 2;
        }
        result += line.trim() + '\\n';
    }
    return result;
}

// Example usage:
var rows = prompt('Enter number of rows:');
var pattern = numberPattern(parseInt(rows));
console.log(pattern);
