swapcase Function:

function swapcase(str=prompt("enter a string")){
    var rstr='';
    for(var i of str){
        if(i.charCodeAt()>=65 && i.charCodeAt()<=90){
            rstr+= i.toLowerCase();
        }
        else if(i.charCodeAt()>=97 && i.charCodeAt()<=122){
            rstr+= i.toUpperCase();
        }
        else{
            rstr+= i;
        }
    }
    return rstr;
}

lowerupper Function:

function lowerupper(a=prompt("enter a string")){
    var l='';
    var u='';
    for(var i of a){
        if(i.toUpperCase() === i){
            l+= i;
        }
        else if(i.toUpperCase() === i){
            u+= i;
        }
    }
    return l + u;
}
