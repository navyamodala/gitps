Pattern Functions in JavaScript

Pattern 1:

        
function pattern(row) {
    var num = 1;
    for (var i = 1; i <= row; i++) {
        var s = '';
        for (var j = 1; j <= i; j++) {
            s += num + " ";
            num++;
        }
        console.log(s);
    }
}
pattern(5);

Alphabet Pattern:

        
function alphapattern(row) {
    for (var i = 1; i <= row; i++) {
        var s = '';
        for (var j = 1; j <= i; j++) {
            s += String.fromCharCode(64 + j) + " ";
        }
        console.log(s);
    }
}
alphapattern(4);

Alphabet Sequence:

        
function alpha(row) {
    var num = 64;
    for (var i = 1; i <= row; i++) {
        var s = '';
        for (var j = 1; j <= i; j++) {
            num++;
            s += String.fromCharCode(num);
        }
        console.log(s);
    }
}
alpha(4);

Pattern with Specific Structure:

        
function patternSpecific(n) {
    for (var i = 1; i <= n; i++) {
        var s = "";
        if (i === 1 || i === n) {
            for (var j = 1; j <= n; j++) {
                s += "* ";
            }
            console.log(s);
        } else {
            for (var j = 1; j <= n; j++) {
                if (j === 1 || j === n) {
                    s += "* ";
                } else {
                    s += "  ";
                }
            }
            console.log(s);
        }
    }
}
patternSpecific(5);

function pattern2(n) {
    for (var i = 1; i <= n; i++) {
        var s = "";
        if (i !== n) {
            for (var j = 1; j <= i; j++) {
                if (j === 1 || j === i) {
                    s += "* ";
                } else {
                    s += "  ";
                }
            }
            console.log(s);
        } else {
            for (var j = 1; j <= i; j++) {
                s += "* ";
            }
            console.log(s);
        }
    }
}

function pattern3(n) {
    for (var i = 1; i <= n; i++) {
        var s = "";
        if (i == n) {
            for (var j = 1; j <= i; j++) {
                s += "* ";
            }
            console.log(s);
        } else if (i == 1) {
            for (var j = 1; j <= n; j++) {
                j == n ? s += "* " : s += "  ";
            }
            console.log(s);
        } else {
            for (var j = 1; j <= n; j++) {
                if (j == n - i + 1 || j == n) {
                    s += "* ";
                } else {
                    s += "  ";
                }
            }
            console.log(s);
        }
    }
}
pattern3(5);
